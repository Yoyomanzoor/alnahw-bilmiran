\begin{english}
    \boldred{\textarabic{المُركَّب الإشارَة}}\index{مركب غير مفيدة!الإشارة} are demonstrative phrases. It consists of the \textarabic{اسم الإِشارَة} (pointer) and \textarabic{مُشار إليه} (what is being pointed to). The \textarabic{مُشار إليه} must be \textarabic{معرفة}.
\end{english}
